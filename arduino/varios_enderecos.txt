#include <TimerOne.h>

char a1[12] = {0x12, 0xAB, 0xA8, 0x37, 0x36, 0x32, 0x38, 0x39, 0x39, 0x30};
char a2[12] = {0x12, 0xAB, 0xB8, 0x35, 0x34, 0x38, 0x35, 0x39, 0x39, 0x30};
char a3[12] = {0x12, 0xAA, 0xA8, 0x37, 0x34, 0x32, 0x35, 0x39, 0x39, 0x30};

char b1[12] = {0x12, 0xAB, 0xA8, 0x37, 0x36, 0x32, 0x38, 0x39, 0x39, 0x31};
char b2[12] = {0x12, 0xAB, 0xB8, 0x35, 0x34, 0x38, 0x35, 0x39, 0x39, 0x31};
char b3[12] = {0x12, 0xAA, 0xA8, 0x37, 0x34, 0x32, 0x35, 0x39, 0x39, 0x31};

bool flag = true;
bool flag0 = true;
void attack() {
  if (flag0) flag0 = false;
  else flag = false;
}

void setup() {
  Serial.begin(115200);
  Timer1.initialize(100 * pow(10, 6));
  Timer1.attachInterrupt(attack);
}

void loop() {
  if (flag) {
    Serial.println(a1);
    Serial.println(a2); 
    Serial.println(a3);
  } else {
    Serial.println(b1);
    Serial.println(b2);
    Serial.println(b3);
    Serial.println(b3);
  }
  delay(10);
}
